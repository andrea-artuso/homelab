auto lo
iface lo inet loopback

auto eno1
iface eno1 inet manual

auto eno2
iface eno2 inet manual

auto eno3
iface eno3 inet manual
	mtu 9000
#SAN interface

auto eno4
iface eno4 inet manual

auto bond0
iface bond0 inet manual
        bond-slaves eno1 eno2
        bond-miimon 100
        bond-mode 802.3ad
        bond-xmit-hash-policy layer2

auto bond0.50
iface bond0.50 inet manual

auto vmbr0
iface vmbr0 inet static
        bridge-ports bond0
        bridge-stp off
        bridge-fd 0
        bridge-vlan-aware yes
        bridge-vids 2-4094

auto vmbr50
iface vmbr50 inet static
        address 10.50.0.1/24
        gateway 10.50.0.254
        bridge-ports bond0.50
        bridge-stp off
        bridge-fd 0

auto vmbr60
iface vmbr60 inet static
	address 10.60.0.1/24
	bridge-ports eno3
	bridge-stp off
	bridge-fd 0
	mtu 9000
#SAN bridge

source /etc/network/interfaces.d/*
